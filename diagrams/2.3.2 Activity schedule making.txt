@startuml Диаграмма активности
title Алгоритм составления сетки расписания врача

start
:Инициализация сетки;
note right: Весь день (00:00-23:59)\nдоступен по умолчанию

:Загрузить стандартное расписание врача;
if (Есть стандартное расписание?) then (да)
  :Ограничить сетку рабочими\nинтервалами\n(например, 09:00-18:00);
else (нет)
  :Установить сетку пустой;
endif

:Загрузить исключения на дату;
fork
  if (Есть доп. смены?) then (да)
    :Добавить доп.смены как\nдоступное время;
  else (нет)
  endif
fork again
  if (Есть отпуска/болезни?) then (да)
    :Отметить дни как\nнедоступные;
  else (нет)
  endif
end fork

:Загрузить существующие записи;
repeat
  :Для каждой записи;
  :Занять слот услуги;
  :Добавить запас ±3 мин\nна подготовку;
repeat while (Есть еще записи\n на указанный день?) is (да)
-> нет;

:Сформировать модель сетки;
:Сериализовать в JSON;
stop
@enduml
